    DLL Injection

    Author : "vinnu"
    Objective : DLL Injection
    OS : Windows based (xp - win7)
    IDE : VC++6.0 (Visual Studio 6)
    Team : Legion Of Xtremers

    Greetz : Team Secfence, Lord DeathStorm, Prashant Uniyal, b0nd, FB1H2S, Garage4Hackers and all of you guys reading this paper.

    In this post I am covering the CreateRemoteThread technique of DLL Injection.

    Actually, this post is being written in the continuity of the "Let us develop Artificial Life" post:

    http://www.garage4hackers.com/showth...rtificial-Life

    But i think, it would be better if, this post gets its own independent thread under the roof of "Exploitation Techniques" forum so that in future we can cover most other types of DLL injections here.

    First of all we need a DLL, So create a new win32 DLL project and write the following code:
    (wait !!!@#$$$RE??~~!! do not copy paste...you guys are boring me...)


    /* dragon.cpp */

    #include "stdafx.h"
    #include "dragon.h"

    void init() {
    MessageBox(NULL,"Dragon Bites", "Blowing Fire==~~~~!",0x00000003);
    }


    BOOL APIENTRY DllMain( HANDLE hModule,
    DWORD ul_reason_for_call,
    LPVOID lpReserved
    )
    {
    switch (ul_reason_for_call)
    {
    case DLL_PROCESS_ATTACH:
    init();break;
    case DLL_THREAD_ATTACH:
    case DLL_THREAD_DETACH:
    case DLL_PROCESS_DETACH:
    break;
    }
    return TRUE;
    }

    /*----------*/


    Copy the compiled DLL in c:\dump folder

    Now we need to develop an injector for this DLL. The injector will grab the handle of the target process and if successfull, then will write the path of the DLL into the target process using the writeProcessMemory function and then ignite a thread using the LoadLibrary function in remote process using CreateRemoteThread.

    This technique works in windows xp/vista/win7 pretty good.

    Following is the code of dll injector, name it injector.cpp and compile the code:


    /* injector.cpp */

    #include <iostream>
    #include <windows.h>
    #include <Tlhelp32.h>

    #define CLIMIT 64
    #define LINTERVAL 10000

    using namespace std;

    int iLoad = 0; // Count of dll injections.

    HANDLE projacker(char *procName) {
    PROCESSENTRY32 pe = { sizeof(PROCESSENTRY32)};
    HANDLE hProcess = NULL;
    char *pointfile = NULL;

    printf("Taking snapshot : ...");
    HANDLE hsnapshot = CreateToolhelp32Snapshot(TH32CS_SNAPALL, 0);
    printf("...Success.\nSearching for Process:...");
    if(Process32First(hsnapshot, &pe)) {
    do {
    printf(".");
    pointfile = CharLower(pe.szExeFile);
    if(!strcmp(pointfile, procName)) {
    printf("Process discovered.\n%s : %d\n", procName, pe.th32ProcessID);
    if(!hProcess) {
    printf("Grabbing Handle : ...");
    return hProcess = OpenProcess(PROCESS_ALL_ACCESS, false, pe.th32ProcessID);
    }
    }Sleep(CLIMIT);
    } while(Process32Next(hsnapshot, &pe));CloseHandle(hsnapshot);hsnapshot = NULL;
    }printf("failed.\n");exit(-1);
    return NULL;
    }

    BOOL dinjector(HANDLE hProcess, LPSTR dllPath) {
    char *ploadLib = "LoadLibraryA";
    char *pLib = "Kernel32";
    HANDLE hmThread = NULL;

    HMODULE hmodKer = GetModuleHandle(pLib);
    if (hmodKer == NULL || hProcess == NULL) return FALSE;
    int dllPathLen = lstrlen (dllPath) + 1;
    LPVOID lpvm = VirtualAllocEx(hProcess, NULL, dllPathLen, MEM_COMMIT, PAGE_READWRITE);
    WriteProcessMemory(hProcess, lpvm, dllPath, dllPathLen, NULL);
    hmThread = CreateRemoteThread(hProcess, NULL, 0, (LPTHREAD_START_ROUTINE)GetProcAddress(hmodKer, ploadLib), lpvm, 0, 0);
    if (hmThread != NULL) {
    iLoad++;
    WaitForSingleObject(hmThread, LINTERVAL);
    }
    VirtualFreeEx(hProcess, lpvm, 0, MEM_RELEASE);
    return true;
    }
    int main (int argc, char* argv[]) {
    if(argc < 2) {
    printf("usage: injector <processname>\n");
    exit(-1);
    }
    char *pdllPath = "c:\\dump\\dragon.dll";
    HANDLE hproc = projacker(argv[1]);
    if(hproc != NULL) {
    printf("...success.\nHandle of %s grabbed\nInjecting DLL : ...%s",argv[1], argv[1]);
    if (dinjector(hproc, pdllPath)) {
    printf("...success.\n");
    CloseHandle(hproc);hproc = NULL;
    } else printf("...failed.\n");exit(1);
    } else printf("...failed.");

    return EXIT_SUCCESS;
    }

    /*-----------end of file---------------*/


    Remember above code is same as projack.cpp in :

    http://www.garage4hackers.com/showth...rtificial-Life

    article with only one difference, instead of terminating the process, we are now injecting the DLL into the process whose handle is grabbed. The DLL is self execution capable with DLL_PROCESS_ATTACH event attached.
